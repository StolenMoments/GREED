# 백테스트 기능 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 룰 기반 주봉 스코어러의 매수 신호를 현재 KOSPI200 구성종목 과거 차트로 이벤트 스터디 백테스트하고, 승률·수익률 분포를 웹에서 조회한다.

**Architecture:** CLI 배치(`scripts/backtest/`)가 FDR로 일봉을 받아 주봉으로 리샘플(기존 `price_bars` interval `"1w"` 캐시 재사용)하고, 지표를 메모리에서 재계산한 뒤, 매 주봉마다 누수 없는 as-of 피처를 재구성해 스코어링한다. 매수 신호의 4·8·12·26주 forward 수익률을 신호별/집계 테이블에 저장하면, 읽기 전용 백엔드 API와 프론트 페이지가 이를 시각화한다.

**Tech Stack:** Python 3.13, pandas, FinanceDataReader, SQLAlchemy(MariaDB), FastAPI, Pydantic v2, React 19 + TypeScript + Vite + @tanstack/react-query + Tailwind.

## 사전 참고 (실행자 필독)

- **DB는 MariaDB**다(SQLite 아님). `DATABASE_URL`이 `.env`에 있어야 백엔드/CLI가 동작한다. 신규 테이블은 `backend/models.py` 모델 + `Base.metadata.create_all`로 생성되며, 기존 패턴에 맞춰 `backend/database.py:_migrate_mariadb()`에도 `CREATE TABLE IF NOT EXISTS`를 추가한다.
- **지표 정의의 단일 출처**는 `scripts/pick.py`다. Task 1에서 이를 `scripts/weekly_indicators.py`로 옮기고 `pick.py`는 재export만 한다. `backend/tests/test_pick_indicators.py`가 `import pick` 후 `pick.add_*`/`pick.append_future_cloud`/`pick.save_csv`를 호출하므로, **이 테스트가 수정 없이 통과**하는 것이 리팩터의 회귀 게이트다.
- **누수(lookahead) 주의**: MA/RSI/MACD/ATR/일목 전환·기준·선행스팬은 인과적(rolling/ewm/`shift(+26)`)이라 안전하다. `ichi_lag`(`shift(-26)`)는 스코어러가 쓰지 않는다. **유일한 미래 누수원은 다이버전스**(`_swing_points`의 `wing=2`가 swing 확정에 미래 2주를 사용)다. Task 4에서 `strict_divergence`에 `+wing` 확정 시프트를 적용해 누수를 제거하고, Task 5의 누수 테스트로 검증한다.
- 테스트 실행: `.venv/Scripts/python.exe -m pytest ...` (Windows venv). 프론트: `cd frontend && npm run build` / 개발 서버 `npm run dev`.

```python
INDICATOR_BUILDERS = (
    add_moving_averages,
    add_liquidity_indicators,
    add_volatility_indicators,
    add_momentum_indicators,
    add_signal_indicators,
    add_ichimoku,
    add_ichimoku_derived_indicators,
)


def add_all_indicators(df: pd.DataFrame) -> pd.DataFrame:
    """주봉 OHLCV df에 전체 지표를 순서대로 추가한다."""
    for builder in INDICATOR_BUILDERS:
        df = builder(df)
    return df


def resample_weekly(df_daily: pd.DataFrame) -> pd.DataFrame:
    """일봉 OHLCV(컬럼 대소문자 무관)를 W-MON 주봉으로 리샘플한다.

    pick.fetch_weekly와 동일한 규칙(label='left', closed='left').
    """
    df = df_daily.copy()
    df.columns = [c.lower() for c in df.columns]
    df["trading_value"] = df["close"] * df["volume"]
    ohlcv = (
        df.resample("W-MON", label="left", closed="left")
        .agg(
            open=("open", "first"),
            high=("high", "max"),
            low=("low", "min"),
            close=("close", "last"),
            volume=("volume", "sum"),
            trading_value=("trading_value", "sum"),
        )
        .dropna(subset=["close"])
    )
    ohlcv.index.name = "date"
    return ohlcv
```

```python
    assert weekly.index.is_monotonic_increasing
    assert weekly["high"].ge(weekly["low"]).all()

    out = wi.add_all_indicators(weekly)
```

```python
def _apply_confirmation_shift(df: pd.DataFrame, wing: int = DIVERGENCE_WING) -> pd.DataFrame:
    """다이버전스는 swing 확정에 미래 wing주를 사용하므로, as-of 누수를 막기
    위해 flag를 wing주 뒤로 이동(확정 시점에만 보이도록)한다. 1회만 적용한다."""
    if df.attrs.get("_div_shifted"):
        return df
    out = df.copy()
    for col in ("strict_divergence", "rsi_divergence", "macd_hist_divergence"):
        if col in out.columns:
            out[col] = out[col].shift(wing)
    out.attrs["_div_shifted"] = True
    return out
```

```python
from __future__ import annotations

import csv
from pathlib import Path

DEFAULT_UNIVERSE_PATH = Path(__file__).resolve().parent / "kospi200.csv"


def load_universe(path: Path | str = DEFAULT_UNIVERSE_PATH) -> list[tuple[str, str]]:
    """`code,name` CSV에서 (6자리 코드, 종목명) 목록을 읽는다."""
    p = Path(path)
    if not p.exists():
        raise FileNotFoundError(
            f"KOSPI200 유니버스 파일이 없습니다: {p}\n"
            "code,name 컬럼을 가진 CSV를 해당 경로에 두세요."
        )
    rows: list[tuple[str, str]] = []
    with p.open(encoding="utf-8-sig", newline="") as f:
        for row in csv.DictReader(f):
            code = str(row.get("code", "")).strip().zfill(6)
            name = str(row.get("name", "")).strip()
            if code.isdigit() and len(code) == 6:
                rows.append((code, name))
    if not rows:
        raise ValueError(f"유니버스 파일에 유효한 종목이 없습니다: {p}")
    return rows
```

```python
import sys

# 워밍업(120) + 일목 선행(26) 여유를 포함한 최소 주봉 수
MIN_HISTORY_WEEKS = 120 + 26 + 60


def _load_cached_weekly(db: Session, ticker: str) -> pd.DataFrame:
    rows = db.scalars(
        select(PriceBar)
        .where(PriceBar.ticker == ticker, PriceBar.interval == WEEKLY_INTERVAL)
        .order_by(PriceBar.bar_date)
    ).all()
    if not rows:
        return pd.DataFrame()
    df = pd.DataFrame(
        {
            "open": [r.open for r in rows],
            "high": [r.high for r in rows],
            "low": [r.low for r in rows],
            "close": [r.close for r in rows],
            "volume": [r.volume for r in rows],
            "trading_value": [r.trading_value for r in rows],
        },
        index=pd.to_datetime([r.bar_date for r in rows]),
    )
    df.index.name = "date"
    return df.dropna(subset=["close"])


def _fetch_daily_max(ticker: str) -> pd.DataFrame | None:
    import FinanceDataReader as fdr

    try:
        df = fdr.DataReader(ticker)  # start 미지정 → 가용 최대 과거
    except Exception:
        return None
    if df is None or df.empty:
        return None
    return df


def _to_upsert_df(weekly: pd.DataFrame) -> pd.DataFrame:
    """resample_weekly 출력(소문자)을 upsert_price_bars가 읽는 대문자 컬럼으로 변환."""
    return pd.DataFrame(
        {
            "Open": weekly["open"],
            "High": weekly["high"],
            "Low": weekly["low"],
            "Close": weekly["close"],
            "Volume": weekly["volume"],
            "TradingValue": weekly["trading_value"],
        },
        index=weekly.index,
    )


def load_weekly_ohlcv(db: Session, ticker: str) -> pd.DataFrame:
    """주봉 OHLCV(소문자 컬럼, date 인덱스)를 반환. 캐시 우선, 부족하면 FDR."""
    cached = _load_cached_weekly(db, ticker)
    if len(cached) >= MIN_HISTORY_WEEKS:
        return cached

    daily = _fetch_daily_max(ticker)
    if daily is None:
        return cached  # 캐시라도 반환(빈 DF일 수 있음)

    weekly = resample_weekly(daily)
    upsert_price_bars(db, ticker, WEEKLY_INTERVAL, _to_upsert_df(weekly))
    return _load_cached_weekly(db, ticker)
```

```python
from __future__ import annotations

import sys
from dataclasses import dataclass, field
from datetime import date
from pathlib import Path

import numpy as np
import pandas as pd

ROOT = Path(__file__).resolve().parent.parent  # scripts/
if str(ROOT) not in sys.path:
    sys.path.insert(0, str(ROOT))

from rule_scorer.features import _apply_confirmation_shift, extract_features_asof  # noqa: E402
from rule_scorer.score import score_features  # noqa: E402
from weekly_indicators import add_all_indicators, append_future_cloud  # noqa: E402

HORIZONS: tuple[int, ...] = (4, 8, 12, 26)
WARMUP_WEEKS = 120
BUCKETS: tuple[str, ...] = ("4-5", "6-7", "8+", "ALL")


@dataclass(slots=True)
class SignalRecord:
    ticker: str
    name: str
    signal_date: date
    score: int
    score_bucket: str
    entry_date: date
    entry_price: float
    returns: dict[int, float | None] = field(default_factory=dict)


@dataclass(slots=True)
class StatRow:
    horizon: int
    score_bucket: str
    count: int
    censored_count: int
    win_rate: float | None
    mean: float | None
    median: float | None
    std: float | None
    p25: float | None
    p75: float | None
    min: float | None
    max: float | None


def score_bucket(total: int) -> str:
    if total >= 8:
        return "8+"
    if total >= 6:
        return "6-7"
    return "4-5"


def _to_date(value) -> date:
    return pd.to_datetime(value).date()


def _f(value) -> float | None:
    if value is None or pd.isna(value):
        return None
    return float(value)


def build_combined(weekly_ohlcv: pd.DataFrame, ticker: str, name: str) -> pd.DataFrame:
    """주봉 OHLCV(date 인덱스, 소문자 컬럼)에 지표·미래구름·식별 컬럼을 붙인
    결합 프레임을 만든다. extract_features_asof가 기대하는 형태."""
    df = add_all_indicators(weekly_ohlcv.copy())
    df = append_future_cloud(df).reset_index()
    df["date"] = df["date"].astype(str).str.slice(0, 10)
    df["ticker"] = ticker
    df["name"] = name
    # 다이버전스 확정 시프트를 여기서 1회 적용해 둔다. extract_features_asof는
    # attrs 가드로 재적용을 건너뛰므로, run_ticker의 매 i 호출에서 프레임 복사를 피한다.
    return _apply_confirmation_shift(df)


def run_ticker(
    combined: pd.DataFrame,
    *,
    horizons: tuple[int, ...] = HORIZONS,
    warmup: int = WARMUP_WEEKS,
) -> list[SignalRecord]:
    price = combined[combined["close"].notna()].reset_index(drop=True)
    n = len(price)
    records: list[SignalRecord] = []
    last_entry_i = n - 2  # 진입은 i+1이므로 i는 n-2까지
    for i in range(warmup, last_entry_i + 1):
        feats = extract_features_asof(combined, i)
        result = score_features(feats)
        if result.judgment != "매수":
            continue
        entry_price = _f(price["open"].iloc[i + 1])
        if entry_price is None or entry_price <= 0:
            continue
        returns: dict[int, float | None] = {}
        for h in horizons:
            j = i + h
            exit_close = _f(price["close"].iloc[j]) if j < n else None
            returns[h] = (exit_close / entry_price - 1) if exit_close else None
        records.append(
            SignalRecord(
                ticker=feats.ticker,
                name=feats.name,
                signal_date=_to_date(price["date"].iloc[i]),
                score=result.total,
                score_bucket=score_bucket(result.total),
                entry_date=_to_date(price["date"].iloc[i + 1]),
                entry_price=entry_price,
                returns=returns,
            )
        )
    return records


def _stat_row(horizon: int, bucket: str, recs: list[SignalRecord]) -> StatRow:
    vals = [r.returns[horizon] for r in recs]
    rets = np.array([v for v in vals if v is not None], dtype=float)
    censored = sum(1 for v in vals if v is None)
    if rets.size == 0:
        return StatRow(horizon, bucket, 0, censored, None, None, None, None, None, None, None, None)
    return StatRow(
        horizon=horizon,
        score_bucket=bucket,
        count=int(rets.size),
        censored_count=censored,
        win_rate=float((rets > 0).mean()),
        mean=float(rets.mean()),
        median=float(np.median(rets)),
        std=float(rets.std(ddof=0)),
        p25=float(np.percentile(rets, 25)),
        p75=float(np.percentile(rets, 75)),
        min=float(rets.min()),
        max=float(rets.max()),
    )


def aggregate(
    records: list[SignalRecord],
    *,
    horizons: tuple[int, ...] = HORIZONS,
) -> list[StatRow]:
    rows: list[StatRow] = []
    for horizon in horizons:
        for bucket in BUCKETS:
            recs = records if bucket == "ALL" else [r for r in records if r.score_bucket == bucket]
            rows.append(_stat_row(horizon, bucket, recs))
    return rows
```

```python
import sys
from datetime import date
from pathlib import Path
```

```python
            last = weekly.index.max().date()
            data_start = first if data_start is None else min(data_start, first)
            data_end = last if data_end is None else max(data_end, last)

        stats = aggregate(all_records)
        run_id = persist_run(

            records=all_records,
            stats=stats,

            data_end=data_end,
            notes=args.notes,

    print(f"\n완료: 종목 {processed}개 / 신호 {len(all_records)}개 / Run ID {run_id}")
```

```python
@router.get("/runs/{run_id}", response_model=BacktestRunDetail)
def get_run(run_id: int, db: Session = Depends(get_db)) -> BacktestRunDetail:
    run = db.get(BacktestRun, run_id)

        raise HTTPException(status_code=404, detail="백테스트 실행을 찾을 수 없습니다.")
    stats = list(

def list_signals(
    run_id: int,

    score_bucket: str | None = None,
    limit: int = Query(100, le=1000),
    offset: int = 0,
    db: Session = Depends(get_db),
) -> BacktestSignalPage:
    if db.get(BacktestRun, run_id) is None:
```

```tsx
          유니버스 {detail.universe} · 종목 {detail.ticker_count}개 · 임계값 {detail.buy_threshold} ·
          기간 {detail.data_start ?? '—'} ~ {detail.data_end ?? '—'}
        </p>
      )}

      {/* 헤드라인: horizon × ALL */}
      {detailLoading ? (
        <div className="h-40 animate-pulse rounded-xl border border-amber-100/10 bg-slate-950/60" />
      ) : (
        <div className="overflow-x-auto">
          <table className="w-full min-w-[640px] border-collapse text-sm">
            <thead>
              <tr className="text-slate-400">
                <th className="px-4 py-2 text-left">보유기간</th>
                <th className="px-4 py-2 text-right">신호 수</th>
                <th className="px-4 py-2 text-right">승률</th>
                <th className="px-4 py-2 text-right">평균</th>
                <th className="px-4 py-2 text-right">중앙값</th>
                <th className="px-4 py-2 text-right">p25 / p75</th>
                <th className="px-4 py-2 text-right">절단</th>
              </tr>
            </thead>
            <tbody>
              {HORIZONS.map((h) => {
                const s = statOf(stats, h, 'ALL');
                return (
                  <tr key={h} className="border-t border-slate-800/70">
                    <td className="px-4 py-2 font-semibold text-slate-200">{h}주</td>
                    <td className="px-4 py-2 text-right text-slate-300">{s?.count ?? 0}</td>
                    <td className="px-4 py-2 text-right text-slate-50">{ratio(s?.win_rate ?? null)}</td>
                    <td className="px-4 py-2 text-right text-emerald-300">{pct(s?.mean ?? null)}</td>
                    <td className="px-4 py-2 text-right text-slate-200">{pct(s?.median ?? null)}</td>
                    <td className="px-4 py-2 text-right text-slate-400">
                      {pct(s?.p25 ?? null)} / {pct(s?.p75 ?? null)}
                    </td>
                    <td className="px-4 py-2 text-right text-slate-500">{s?.censored_count ?? 0}</td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>
      )}

      {/* 점수 구간 비교 (4주 기준 승률) */}
      {!detailLoading && stats.length > 0 && (
        <div className="flex flex-col gap-3">
          <h3 className="text-sm font-semibold text-slate-300">점수 구간별 승률 (보유 4주)</h3>
          <div className="flex flex-col gap-2">
            {BUCKETS.map((b) => {
              const s = statOf(stats, 4, b);
              const wr = s?.win_rate ?? null;
              return (
                <div key={b} className="flex items-center gap-3">
                  <span className="w-12 text-xs text-slate-400">{b}</span>
                  <div className="h-4 flex-1 overflow-hidden rounded bg-slate-800">
                    <div
                      className="h-full bg-amber-400"
                      style={{ width: wr === null ? '0%' : `${wr * 100}%` }}
                    />
                  </div>
                  <span className="w-24 text-right text-xs text-slate-300">
                    {ratio(wr)} (n={s?.count ?? 0})
                  </span>
                </div>
              );
            })}
          </div>
        </div>
      )}
    </section>
  );
}

export default BacktestPage;
```
